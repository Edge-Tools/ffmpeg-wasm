# Meson cross file for cross-compiling to WebAssembly with emscripten.
# Used by the codec libraries whose build system is meson (dav1d, harfbuzz,
# fribidi). The -pthread / -msimd128 flags match the rest of the build so every
# object links cleanly into the threaded ffmpeg module.
[binaries]
c = 'emcc'
cpp = 'em++'
ar = 'emar'
ranlib = 'emranlib'
strip = 'emstrip'
pkg-config = 'pkg-config'

[built-in options]
c_args = ['-O3', '-msimd128', '-pthread']
cpp_args = ['-O3', '-msimd128', '-pthread']
c_link_args = ['-O3', '-pthread']
cpp_link_args = ['-O3', '-pthread']

[host_machine]
system = 'emscripten'
cpu_family = 'wasm32'
cpu = 'wasm32'
endian = 'little'
